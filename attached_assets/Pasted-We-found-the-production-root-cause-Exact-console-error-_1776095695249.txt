   Normalize invalid values to "pass_to_customer".

Add something like:

type FeeMode = "pass_to_customer" | "absorb" | "split";

function normalizeFeeMode(mode: unknown): FeeMode {
if (
mode === "pass_to_customer" ||
mode === "absorb" ||
mode === "split"
) {
return mode;
}
return "pass_to_customer";
}

Then use normalized feeMode inside calculateBookingPricing instead of throwing on invalid input.
